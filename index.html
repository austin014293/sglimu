💡

PIEV 핵심 모델

  • P (인지 / Perception): GPS 위치 인지 및 자이로 가속도 차량 상태 감지
  • I (판단 / Intellection): ESP32 제어기를 통해 터널 진입 전 "점등 필수" 판단
  • E (감정 / Emotion): 진입 시 암흑 공포 완벽 해소
  • V (실행 / Volition): MOSFET 스위칭으로 0초 만에 미등 즉시 구동
한 줄 요약: "오토라이트의 3~5초 딜레이를 GPS 예측 기술로 극복하여 터널 진입 순간의 안전 공백을 0초로 만드는 기술"

발전 과정: 2. GPS + 자이로/가속도 센서 퓨전

1) 가속도 정적 오프셋(보정치) 적용

자이로/가속도 센서 자체의 정적 제조 오차가 상시 주행 신호로 잡혀 오독함.

  • 정지 상태에서 데이터 평균 수집 ➔ Calibration 오프셋 상수 도출
  • BIAS_X(2.47), BIAS_Y(-1.53), BIAS_Z(1.19)를 적용하여 정지 시 영점 보정

2) 주행 상태 필터링 (Motion Filter)

중력가속도를 제외한 차량의 실제 움직임 크기(motionMag)를 기하학적으로 연산.

  • 합성 수식: $\sqrt{X^2 + Y^2 + Z^2}$ 결과에서 중력(9.8) 상쇄
  • 임계값(0.6) 이상 진동 시에만 '주행 중(isMoving=true)'으로 판정
  • 정지 시 GPS 신호 무시 및 10차 이동평균 필터 갱신 차단

⚠️ 경계부 시스템 채터링 해결

조도 기준값(임계값)을 단일값으로 지정할 경우, 그늘이나 경계 영역에서 조도값이 조금만 흔들려도 미등이 켜졌다 꺼지는 채터링(Chattering) 현상 발생.

  • 해결책: 듀얼 임계값(Hysteresis) 설계
  • 낮 판별: 조도 3500 이상 2초 유지 시 낮으로 전환
  • 밤 판별: 조도 500 이하 2초 유지 시 밤으로 전환

  • 맑은 낮 실외: 약 50,000 lx | 그늘진 실외: 약 3,000 lx
  • 터널 실내: 약 10~50 lx | 형광등 실내: 약 600 lx

ESP32 임베디드 핵심 소스코드

📄 esp32_firmware.ino
COPY CODE
#include <TinyGPS++.h>
#include <Adafruit_MPU6050.h>
#include <Adafruit_Sensor.h>
#include <Wire.h>
#include <BH1750.h> 

TinyGPSPlus gps;
Adafruit_MPU6050 mpu;
HardwareSerial gpsSerial(2); 
BH1750 lightMeter; 

// ==========================================
// [Sector 1] 변수 선언 및 타겟 설정
// ==========================================
const double TARGET_LAT = 35.105925;
const double TARGET_LNG = 129.007105;
const float BIAS_X = 2.47;
const float BIAS_Y = -1.53;
const float BIAS_Z = 1.19;
const int ledPin = 25;  
const int ledChannel = 0;
const int ledFreq = 5000;
const int ledRes = 8; 
const float NIGHT_LIMIT = 800.0; 
bool mpuReady = false;
const int AVG_SIZE = 10;
double latBuffer[AVG_SIZE], lngBuffer[AVG_SIZE];
int bufIdx = 0;
double lastValidLat = 0, lastValidLng = 0;
bool firstFix = false;

// ==========================================
// 하드웨어 초기화 및 세팅
// ==========================================
void setup() {
    Serial.begin(115200); 
    delay(1500);
    #if ESP_ARDUINO_VERSION_MAJOR < 3
    ledcSetup(ledChannel, ledFreq, ledRes);
    ledcAttachPin(ledPin, ledChannel);
    #else
    ledcAttach(ledPin, ledFreq, ledRes);
    #endif
    gpsSerial.begin(9600, SERIAL_8N1, 16, 17); 
    Wire.begin(21, 22);                        
    lightMeter.begin();
    if (mpu.begin()) {
        mpuReady = true;
        Serial.println("가속도센서(MPU6050) 필터링 준비 완료");
    } else {
        Serial.println("가속도센서연결실패! 필터링없이진행합니다.");
    }
}

void loop() {
    unsigned long currentTime = millis(); 
    while (gpsSerial.available() > 0) {
        gps.encode(gpsSerial.read());
    }
}
// ==========================================
// [Sector 2] 1차 물리필터: 가속도기반 주행감지
// ==========================================
bool isMoving = true; 
if (mpuReady) {
    sensors_event_t a, g, temp;
    mpu.getEvent(&a, &g, &temp);
    float adjX = a.acceleration.x - BIAS_X;
    float adjY = a.acceleration.y - BIAS_Y;
    float adjZ = a.acceleration.z - BIAS_Z;
    float motionMag = sqrt(adjX*adjX + adjY*adjY + (adjZ - 9.8)*(adjZ - 9.8));
    isMoving = (motionMag > 0.6);
}

// ==========================================
// [Sector 3] 2차 위치융합: GPS 이동평균 및 거리 연산
// ==========================================
double distanceToTarget = 9999.0; 
if (gps.location.isValid() && gps.satellites.value() >= 4) {
    double rawLat = gps.location.lat();
    double rawLng = gps.location.lng();
    if (!firstFix) {
        for(int i=0; i<AVG_SIZE; i++) { 
            latBuffer[i] = rawLat; 
            lngBuffer[i] = rawLng; 
        }
        firstFix = true;
        lastValidLat = rawLat; 
        lastValidLng = rawLng;
    } 
    else if (isMoving) {
        latBuffer[bufIdx] = rawLat; 
        lngBuffer[bufIdx] = rawLng;
        bufIdx = (bufIdx + 1) % AVG_SIZE;
        double sumLat = 0, sumLng = 0;
        for(int i=0; i<AVG_SIZE; i++) { 
            sumLat += latBuffer[i]; 
            sumLng += lngBuffer[i]; 
        }
        lastValidLat = sumLat / AVG_SIZE; 
        lastValidLng = sumLng / AVG_SIZE;
    }
}
if (firstFix) {
    distanceToTarget = TinyGPSPlus::distanceBetween(lastValidLat, lastValidLng, TARGET_LAT, TARGET_LNG);
}
// ==========================================
// [Sector 4] 조건분리형 스마트 미등제어(조도센서 측정 포함)
// ==========================================
float lux = lightMeter.readLightLevel();
int brightness = 0;
if (lux <= NIGHT_LIMIT && lux >= 0) {
    brightness = 255; 
} 
else {
    if (!firstFix || distanceToTarget >= 300.0) {
        brightness = 0;    
    } 
    else if (distanceToTarget < 300.0 && distanceToTarget > 45.0) {
        brightness = (int)((300.0 - distanceToTarget) * (255.0 / (300.0 - 45.0)));
    } 
    else if (distanceToTarget <= 45.0) {
        brightness = 255;  
    }
}

// ==========================================
// [Sector 5-B] 하드웨어 드라이빙 및 모니터링 출력
// ==========================================
brightness = constrain(brightness, 0, 255);
#if ESP_ARDUINO_VERSION_MAJOR < 3
ledcWrite(ledChannel, brightness);
#else
ledcWrite(ledPin, brightness); 
#endif

static unsigned long lastPrint = 0;
if (currentTime - lastPrint > 1000) {
    if (lux <= NIGHT_LIMIT && lux >= 0) {
        Serial.print("[야간모드-상시점등]");
    } else {
        Serial.print("[주간모드-거리제어]");
    }
    Serial.print(" 조도: "); Serial.print(lux, 0);
    if (firstFix) {
        Serial.print("lx | 거리: "); 
        Serial.print(distanceToTarget, 1); Serial.print("m");
    } else {
        Serial.print("lx | 거리: 위성 대기 중...");
    }
    Serial.print(" | 미등 밝기: "); 
    Serial.print((int)((brightness / 255.0) * 100)); 
    Serial.print("%");
    Serial.print(" (PWM: "); Serial.print(brightness); 
    Serial.println(")");
    lastPrint = currentTime;
}

CAN 통신 제어 시도 및 분석

1) B-CAN (Body CAN) 제어 원리

바디 전장 시스템(미등/헤드라이트, 도어록, 와이퍼 등)을 제어하기 위한 차량 내 통신 규격.

  • 꼬임선(Twisted Pair: CAN_High, CAN_Low) 구조
  • 노이즈 방지를 위해 차동 전압(Differential Voltage) 사용

2) SGW (보안 게이트웨이) 방화벽의 한계

테스트 차량인 아반떼 CN7 22년식 기준, OBD2 포트를 통한 패킷 주입 실패.

  • 차량 보안(해킹 방지)을 위해 보안 게이트웨이(SGW) 방화벽 탑재
  • 진단 도구 접속 시 신호 필터링 차단
📸 B-CAN 대기 전압 (2.56V)
📸 B-CAN 구동 전압 (2.49V)

CAN 통신 제어 실험 성공 및 한계 극복

SGW 방화벽을 우회하기 위해 정션박스 후면의 B-CAN 직접 T-tap 배선 접촉을 시도하여 통신 수신에 성공하였습니다.

T-tap 수신 및 데이터 필터링

  • 정션박스 다이렉트 탭을 통해 총 33개의 활성 CAN ID 수신
  • 라이트 레버 조작 시 변동 데이터 추적 ➔ ID 0x168 변동 감지
  • 미등 OFF: 0x01 (이진수: 0000 0001) / 미등 ON: 0x89 (이진수: 1000 1001)

명령 주입 충돌 및 ICU 에러 처리

  • 순정 BCM이 주간에 10~20ms 주기로 '미등 OFF' 패킷 전송 중
  • ESP32 모듈이 5ms 간격으로 '미등 ON' 패킷을 주입
  • 비트 충돌로 전압 감쇄 발생 ➔ ICU가 Bit Error(통신 에러)로 판정하여 보호 모드 전환

법규 기준 내 자동 제어 설계 조건 검토 결과: **합법 범위 충족**

🙋 Q&A 2. GPS 드리프트 오차 보정

🛰️ GPS 수신 특성 및 드리프트 오차 제어

차량이 정지 상태이거나 터널 진입 전 신호 감쇄가 발생할 때 GPS 좌표가 사방으로 튀는 드리프트(Drift) 오차를 억제하기 위한 핵심 메커니즘입니다.

  • 속도 기반 임계치 설정: 속도가 0.5km/h 이하일 경우 위치 업데이트를 제한하고 이전 신뢰값을 유지합니다.
  • HDOP(수평 정밀도 저하율) 모니터링: HDOP 값이 기준치(예: 2.0) 이상으로 저하되는 경우 터널 매칭 로직에서 해당 신호를 제외합니다.
  • 방향 벡터 보정: 자이로 센서의 방위각 데이터를 활용하여 GPS 궤적과 차량 진행 방향의 평행 여부를 대조 검증합니다.

🙋 Q&A 3. 자이로 MPU6050 오프셋 캘리브레이션

🔄 가속도·자이로 센서 융합 및 초기 오프셋 제거

터널 진입 각도 및 차량의 움직임을 실시간으로 추적하기 위해 MPU6050 센서의 초기 누적 오차(오프셋)를 제어하고 센서 데이터를 필터링하는 기법입니다.

  • 초기화 캘리브레이션: 부팅 시 첫 2~3초간 차량이 정지한 상태에서 200회 이상의 샘플을 채집하여 X, Y, Z축의 자이로/가속도 평균 편차를 초기 오프셋 값으로 저장 및 차감합니다.
  • 보색 필터(Complementary Filter) 적용: 고주파 노이즈에 약한 가속도계 데이터와 적분 오차(드리프트)가 있는 자이로 데이터를 Angle = 0.98 * (Angle + Gyro * dt) + 0.02 * (AccelAngle) 수식으로 융합하여 빠르고 안정적인 반응성을 구현했습니다.

🙋 Q&A 4. 제조사 OTA(무선 업데이트)만으로 활성화 가능성

✅ OTA 펌웨어 업데이트만으로 미등 제어가 가능한 환경과 확장성

최신 차량 및 소프트웨어 중심 자동차(SDV) 환경에서는 하드웨어 변경 없이 OTA 소프트웨어 업데이트만으로 이 안전 제어 기능을 즉시 구현 및 배포할 수 있습니다.

  • 최신 차량의 하드웨어 준비성 (PWM IPS 및 LED 기본 탑재): 최근 출시되는 대부분의 신차(국내 등록 차량의 약 20~30%)는 BCM(바디제어모듈)과 미등 라인 사이에 고정밀 PWM 디밍이 가능한 지능형 반도체 스위치(IPS)를 탑재하고 있으며, 미등 및 제동등 역시 LED로 설계되어 있습니다. 이 경우 추가 하드웨어 없이 소프트웨어(BCM 펌웨어) 업데이트만으로 터널 진입 전 선제 점등 및 디밍 로직을 즉시 활성화할 수 있습니다.
  • 무선 업데이트(OTA)를 통한 대규모 신속 보급: 2020년 국내 무선 업데이트 규제 완화 및 제도 안착에 따라 출고된 SDV 지향 차량들은 무선 펌웨어 업데이트를 기본 지원합니다. 제조사가 개발한 능동 제어 로직을 BCM 펌웨어에 포함하여 OTA로 배포할 시, **정비소 방문 없이 수백만 대의 차량에 안전 기능을 즉시 전파**할 수 있습니다.
  • 본 프로젝트 제안의 확장성 (이중 트랙 솔루션):
    • 소프트웨어 트랙 (신차): 본 프로젝트에서 설계·검증한 GPS/자이로 센서 및 CAN 통신 데이터 융합 알고리즘은 신차의 BCM 내장 소프트웨어 알고리즘으로 그대로 이식되어 OTA 방식으로 대량 배포가 가능합니다.
    • 하드웨어 트랙 (기존 노후/보급 차량): BCM PWM 디밍 하드웨어가 결여된 기계식 릴레이 및 할로겐 미등 차량(전체 차량의 약 70%, 평균 차령 15.6년)에 대해서는 본 연구에서 개발한 외장형 MOSFET 디밍 제어 보드와 우회 배선(T-tap) 방식을 통해 저비용으로 완벽하게 기능을 지원합니다.
  • 결론: 본 제안 시스템은 신차군에는 **소프트웨어 펌웨어(OTA) 솔루션**으로, 기존 운행차군에는 **하드웨어 모듈 솔루션**으로 동시에 대응하여 대한민국 도로 위 모든 차량에 빠르게 적용 가능한 **하이브리드형 안전 기술 표준**입니다.

🙋 Q&A 5. 기존 오토라이트(조도 센서) 대비 GPS 제어의 필요성

🛰️ 반응형(Reactive) 제어 vs 예측형(Proactive) 제어

기존 차량용 조도 센서 기반 오토라이트 기능이 있음에도, GPS 기반의 선제 제어 시스템이 유발하는 높은 사고 방지 효과와 차별성을 설명합니다.

  • 지연 반응 속도의 한계 (오토라이트의 약 3~5초 딜레이): 오토라이트는 가로수 그늘이나 임시 조도 하락에 반응하여 라이트가 계속 켜지고 꺼지는 오작동을 차단하기 위해 3~5초의 하드웨어/소프트웨어 필터 딜레이를 가지고 있습니다. 시속 100km/h로 주행 시 이 딜레이는 차량이 약 83m를 미등 없이 달리게 만들어 터널 입구의 블랙홀 사고 구간을 유발합니다.
  • 터널 진입 전 사전 점등 완료 (Pre-lighting): 본 GPS 시스템은 터널 포털(진입구) 좌표를 사전에 파악하여 진입부 **50~100m 전방에서 이미 미등을 100% 밝기로 점등**합니다. 터널 안으로 진입하는 순간 미등이 이미 켜져 있으므로 뒤따라오는 후방 차량에 즉각 경고를 주어 급제동 및 추돌 사고를 원천 차단합니다.
  • 안전 성능 요약: 조도 센서처럼 어두워진 '후'에 동작하는 수동적 제어가 아닌, 터널이라는 지리적 객체를 인식해 미리 대처하는 **선제적 능동 안전 제어로 후방 추돌 예방률을 획기적으로 상승**시킵니다.

🙋 Q&A 6. MOSFET PWM 디밍 발열 제어 및 회로 안정성

⚡ TC4427 게이트 드라이버 도입을 통한 발열 및 노이즈 해결

고속 PWM 스위칭 환경에서 발생할 수 있는 MOSFET 소자의 물리적 손상 및 열 누적 문제를 전기/전자공학적으로 해결한 설계 기법입니다.

  • MCU 다이렉트 구동의 한계 극복 (TC4427 필수 적용): ESP32 등 일반 MCU의 입출력 핀 전류 공급 한계(약 20mA)로는 N-Channel MOSFET의 게이트 입력 용량(Gate Capacitance)을 충분히 신속하게 충전시킬 수 없어 스위칭 과도 구간(Linear Region)이 길어지고 극심한 열손실이 발생합니다. 순간 1.5A를 뿜어내는 **게이트 드라이버 TC4427**을 내장하여 충·방전 딜레이를 최소화함으로써 열 누적을 완벽히 방지했습니다.
  • 서지 스파이크 및 유도성 부하(Inductive Load) 대책: 차량용 순정 벌브 및 배선은 코일 성분(유도성 부하)을 지니고 있어 PWM 스위칭 차단 시 높은 역기전력 전압 스파이크가 발생해 MOSFET을 파괴할 수 있습니다. D-S 단에 보호용 플라이백 다이오드를 결선하여 전압 역류를 안전하게 그라운드로 흡수시켰습니다.

🙋 Q&A 7. 현대자동차 전장부서 공식 질의 회신

📄 현대자동차·기아 전장 개발팀 공식 서면 회신 내용

본 과제의 기술적 타당성과 법규 부합성에 대하여 현대자동차·기아 전장 부서에 직접 공식 질의하고 회신받은 원본 자료입니다.

  • 법규 적합성 유권 해석 완료: 미등 밝기 조절은 '법규 허용 범위 내 자동 제어'이므로 **명백한 합법**이며, 시스템 자동 제어에 의한 단계적 감광은 현행 양산 차량에 이미 탑재되고 설계 규정에 부합하는 방식임을 확인받았습니다.
  • BCM 및 B-CAN 제어 원리 검증: B-CAN 통신을 이용한 미등 밝기 제어 명령 전달 및 BCM의 IPS를 통한 PWM 구동 제어 구조가 양산형 조명 제어 표준 기술임을 확인받았습니다.
  • 하드웨어 한계 및 펌웨어 업데이트 유효성 확인: IPS 소자가 장착된 최신 차량이라도 BCM 하드웨어·배선·전원 사양이 막혀 있으면 펌웨어(소프트웨어) 업데이트만으로는 제어가 불가능하여 **외장형 하드웨어 추가 회로 장착(본 설계 결과물)이 필수적**이라는 기술적 답변을 확보했습니다.
